// Smart Detection Update Implementation

public function syncScopus(string $issn): void
{
    $live = $this->api->fetch($issn);
    $db = $this->dao->getMetrics($issn);

    // Write ONLY if CiteScore changes
    if ($live['citescore'] !== $db['citescore']) {
        $this->dao->update($live);
        $this->logger->info('Metrics updated');
    }
}
